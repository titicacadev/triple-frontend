# TripleWeb

## Usage

```tsx
// app/layout.tsx
import { TripleWeb, buildTripleWebProps } from '@titicaca/triple-web-nextjs'
import i18n from '../src/i18n'

export default async function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html>
      <body>
        <TripleWeb
          {...(await buildTripleWebProps())}
          envProvider={{ ... }}
          i18nProvider={{ i18n, fallback: 'ko'}}
        >
          {children}
        </TripleWeb>
      </body>
    </html>
  )
}
```

### children
