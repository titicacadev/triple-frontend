# TripleWeb

## Usage

```tsx
// pages/_app.tsx
import { TripleWeb, buildTripleWebProps, BuildTripleWebPropsResult } from '@titicaca/triple-web-nextjs-pages'

export default function MyApp({
  Component,
  pageProps,
  clientAppProvider,
  sessionProvider,
  userAgentProvider,
}: AppProps & BuildTripleWebPropsResult) {
  return (
    <TripleWeb
      clientAppProvider={clientAppProvider}
      envProvider={{ ... }}
      i18nProvider={{ ... }}
      sessionProvider={sessionProvider}
      userAgentProvider={userAgentProvider}
    >
      <Component {...pageProps} />
    </TripleWeb>
  )
}

MyApp.getInitialProps = async (
  appContext: AppContext
): Promise<AppInitialProps & BuildTripleWebPropsResult> => {
  const { ctx } = appContext

  const appInitialProps = await App.getInitialProps(appContext)

  return {
    ...appInitialProps,
    ...(await buildTripleWebProps(ctx)),
  }
}
```
